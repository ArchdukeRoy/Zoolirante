CREATE TABLE People (
personID int identity(10000,1) PRIMARY KEY,
firstName nvarchar(100) NOT NULL,
lastName nvarchar(100) NOT NULL,
email nvarchar(255) NOT NULL unique
);

CREATE TABLE Roles (
roleID   int identity(1000,1) PRIMARY KEY,
roleName nvarchar(100) NOT NULL unique
);

CREATE TABLE Staff (
staffID int PRIMARY KEY,
mobile nvarchar(50) NULL,
photo nvarchar(255) NULL,
password nvarchar(255) NOT NULL,
roleID int NOT NULL,
CONSTRAINT FK_Staff_People FOREIGN KEY (staffID) REFERENCES People(personID),
CONSTRAINT FK_Staff_Roles FOREIGN KEY (roleID)  REFERENCES Roles(roleID)
);

CREATE TABLE Admin (
adminID int PRIMARY KEY,
CONSTRAINT FK_Admin_Staff FOREIGN KEY (adminID) REFERENCES Staff(staffID)
);

CREATE TABLE ZooKeeper (
zookeeperID int PRIMARY KEY,
description nvarchar(MAX) NOT NULL,
CONSTRAINT FK_ZK_Staff FOREIGN KEY (zookeeperID) REFERENCES Staff(staffID)
);

CREATE TABLE Events (
eventID int identity (1,1) PRIMARY KEY,
name nvarchar(200) NOT NULL,
description nvarchar(MAX) NULL
);

CREATE TABLE EventRollCall (
eventRollCallID int identity(1,1) PRIMARY KEY,
rollDate date NOT NULL,
rollTime time NOT NULL,
zookeeperID int,
eventID int NOT NULL,
CONSTRAINT FK_RollCall_ZK FOREIGN KEY (zookeeperID) REFERENCES ZooKeeper(zookeeperID),
CONSTRAINT FK_RollCall_Evt FOREIGN KEY (eventID) REFERENCES Events(eventID),
CONSTRAINT UQ_RollCall UNIQUE (eventID, rollDate, rollTime)
);

CREATE TABLE Visitors (
visitorID int PRIMARY KEY,
username nvarchar(100) NOT NULL unique,
passwordHash nvarchar(255) NOT NULL,
contact nvarchar(100) NULL,
paymentMethod nvarchar(100) NULL,
CONSTRAINT FK_Visitors_People FOREIGN KEY (visitorID) REFERENCES People(personID)
);

CREATE TABLE Species (
speciesID int identity(1,1) PRIMARY KEY,
name nvarchar(200) NOT NULL UNIQUE,
speciesImage nvarchar(500) NULL
);

CREATE TABLE Animals (
animalID int identity(1,1) PRIMARY KEY,
name nvarchar(200) NOT NULL,
gender CHAR(1) NOT NULL check (gender IN ('M','F')),
age int NOT NULL,
description nvarchar(1000) NULL,
speciesID int NOT NULL,
animalImage  nvarchar(500) NULL,
CONSTRAINT FK_Animals_Species FOREIGN KEY (speciesID) REFERENCES Species(speciesID)
);

CREATE TABLE FavouriteAnimals (
favAnimalsID int IDENTITY(1,1) PRIMARY KEY,
visitorID int NOT NULL,
animalID int NOT NULL,
CONSTRAINT FK_Fav_V FOREIGN KEY (visitorID) REFERENCES Visitors(visitorID) ON DELETE CASCADE,
CONSTRAINT FK_Fav_A FOREIGN KEY (animalID) REFERENCES Animals(animalID) ON DELETE CASCADE,
CONSTRAINT UQ_Fav_Visitor_Animal UNIQUE (visitorID, animalID)
);

CREATE TABLE PurchaseHistory (
purchaseHistoryID int PRIMARY KEY,
CONSTRAINT FK_PH_Visitor FOREIGN KEY (purchaseHistoryID)
REFERENCES Visitors(visitorID) ON DELETE CASCADE
);

CREATE TABLE Receipts (
receiptID int identity(1,1) PRIMARY KEY,
receiptDetails NVARCHAR(MAX) NULL,
dateTime DATETIME2 NOT NULL DEFAULT SYSUTCDATETIME(),
purchaseHistoryID int NOT NULL,
CONSTRAINT FK_Receipt_PH FOREIGN KEY (purchaseHistoryID) REFERENCES PurchaseHistory(purchaseHistoryID) ON DELETE CASCADE
);

CREATE TABLE Tickets (
ticketID int identity(1,1) PRIMARY KEY,
dateOfEntry date NOT NULL,
datePaid datetime NULL,
visitorID int NOT NULL,
CONSTRAINT FK_Tickets_Visitor FOREIGN KEY (visitorID) REFERENCES Visitors(visitorID) ON DELETE CASCADE
);

CREATE TABLE Merchandise (
itemID int identity(1,1) PRIMARY KEY,
itemName nvarchar(200) NOT NULL,
itemDescription nvarchar(2000) NULL,
itemCost decimal(10,2) NOT NULL,
itemImage nvarchar(500) NULL
);

CREATE TABLE VisitorMerchOrder (
orderNumber int identity(1,1) PRIMARY KEY,
orderDate datetime NOT NULL DEFAULT SYSUTCDATETIME(),
datePaid datetime NULL,
visitorID int NOT NULL,
CONSTRAINT FK_Order_Visitor FOREIGN KEY (visitorID) REFERENCES Visitors(visitorID) ON DELETE CASCADE
);

CREATE TABLE MerchInOrder (
orderNumber int NOT NULL,
itemID int NOT NULL,
quantity int NOT NULL,
unitPrice decimal(10,2) NOT NULL,
PRIMARY KEY (orderNumber, itemID),
CONSTRAINT FK_MIO_Order FOREIGN KEY (orderNumber) REFERENCES VisitorMerchOrder(orderNumber) ON DELETE CASCADE,
CONSTRAINT FK_MIO_Item FOREIGN KEY (itemID) REFERENCES Merchandise(itemID)
);

CREATE INDEX IX_Staff_roleID ON Staff(roleID);
CREATE INDEX IX_RollCall_eventID ON EventRollCall(eventID);
CREATE INDEX IX_Animals_speciesID ON Animals(speciesID);
CREATE INDEX IX_Fav_Visitor ON FavouriteAnimals(visitorID);
CREATE INDEX IX_Fav_Animal ON FavouriteAnimals(animalID);
CREATE INDEX IX_Receipts_ph ON Receipts(purchaseHistoryID);
CREATE INDEX IX_Tickets_visitorID ON Tickets(visitorID);
CREATE INDEX IX_Order_visitorID ON VisitorMerchOrder(visitorID);
